249 float DHT::CalcdewPoint(float celsius, float humidity) {
250  float A0= 373.15/(273.15 + celsius);
251  float SUM = -7.90298 * (A0-1);
252  SUM += 5.02808 * log10(A0);
253  SUM += -1.3816e-7 * (pow(10, (11.344*(1-1/A0)))-1) ;
254  SUM += 8.1328e-3 * (pow(10,(-3.49149*(A0-1)))-1) ;
255  SUM += log10(1013.246);
256  float VP = pow(10, SUM-3) * humidity;
257  float T = log(VP/0.61078); // temp var
258  return (241.88 * T) / (17.558-T);
259 }
